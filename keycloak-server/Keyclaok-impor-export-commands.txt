	@Hook(Pointcut.SERVER_INCOMING_REQUEST_PRE_PROCESSED)
	public boolean modifyUrl(HttpServletRequest theRequest, HttpServletResponse theResponse) {
		String originalUrl = theRequest.getRequestURL().toString();
		String queryString = theRequest.getQueryString();
		System.out.println("original URL: " + originalUrl);
		System.out.println("queryString URL: " + queryString);

		if (queryString != null && queryString.contains("|")) {
			String modifiedQueryString = queryString.replace("|", "%7C");
			String modifiedUrl = originalUrl + "?" + modifiedQueryString;
			System.out.println("Modified URL: " + modifiedUrl);
			theRequest.setAttribute("javax.servlet.forward.query_string", modifiedQueryString);
		}
		return true;
	}
